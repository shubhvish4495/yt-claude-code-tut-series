/statusline add status line which shows current model being used along with context 
percentage color coded


or directly paste this in ~/.claude/settings.json 
{
  "statusLine": {
    "type": "command",
    "command": "input=$(cat); model=$(echo \"$input\" | jq -r '.model.display_name'); ctx=$(echo \"$input\" | jq -r '.context_window.used_percentage // 0'); if (( $(echo \"$ctx < 61\" | bc -l) )); then c=\"\\033[32m\"; elif (( $(echo \"$ctx < 86\" | bc -l) )); then c=\"\\033[33m\"; else c=\"\\033[31m\"; fi; r=\"\\033[0m\"; branch=$(git rev-parse --abbrev-ref HEAD 2>/dev/null); branch_str=${branch:+\" \\033[36m($branch)\\033[0m\"}; printf \"${c}%s${r} | Context: ${c}%d%%${r}%b\" \"$model\" \"$ctx\" \"$branch_str\""
  },
  "model": "claude-sonnet-4-6[1m]"
}
